[27-Jun-2023 13:31:31 Europe/Berlin] SELECT dettaglio_spedizioni.TITOLO,dettaglio_spedizioni.QUANTITA,
            titoli.EDITORE AS EDITORE_TITOLO,
            titoli.CODICE_ISBN  AS CODICE_TITOLO,
            titoli.TITOLO  AS NOME_TITOLO ,
       (SELECT gruppi_case_ed.DESCRIZIONE
          FROM gruppi_x_case_ed
                   JOIN case_editrici_amiche ON (case_editrici_amiche.CHIAVE = gruppi_x_case_ed.CASA_ED)
                   JOIN gruppi_case_ed ON (gruppi_case_ed.CHIAVE = gruppi_x_case_ed.GRUPPO)
                   JOIN province_x_gruppi_editoriali ON (province_x_gruppi_editoriali.GRUPPO = gruppi_x_case_ed.GRUPPO)
        WHERE gruppi_case_ed.RIVALE = 'F' AND
              istituti.PROVINCIA = province_x_gruppi_editoriali.PROVINCIA AND
                   case_editrici_amiche.DESCRIZIONE IN (SELECT titoli.EDITORE 
                                                                             FROM titoli 
                                                                           WHERE CHIAVE=dettaglio_spedizioni.TITOLO)
		  LIMIT 0,1) AS GRUPPO_CASA,
  FROM dettaglio_spedizioni LEFT OUTER JOIN titoli ON (titoli.CHIAVE = dettaglio_spedizioni.TITOLO)
       JOIN spedizioni ON (spedizioni.CHIAVE = dettaglio_spedizioni.SPEDIZIONE)
       JOIN istituti ON (istituti.CHIAVE = spedizioni.ISTITUTO)
 WHERE STATO='C' 
   AND dettaglio_spedizioni.DATA_ULTIMA_MODIFICA >= '2023-05-28' 
   AND dettaglio_spedizioni.DATA_ULTIMA_MODIFICA <= '2023-06-28' 
       AND (SELECT gruppi_case_ed.CHIAVE
                                  FROM gruppi_x_case_ed
			JOIN case_editrici_amiche ON (case_editrici_amiche.CHIAVE = gruppi_x_case_ed.CASA_ED)
			JOIN gruppi_case_ed ON (gruppi_case_ed.CHIAVE = gruppi_x_case_ed.GRUPPO)
			JOIN province_x_gruppi_editoriali ON (province_x_gruppi_editoriali.GRUPPO = gruppi_x_case_ed.GRUPPO)
		    WHERE gruppi_case_ed.RIVALE = 'F' AND
		                                     istituti.PROVINCIA = province_x_gruppi_editoriali.PROVINCIA AND
                                           case_editrici_amiche.DESCRIZIONE IN (SELECT titoli.EDITORE 
	                                                                                        FROM titoli 
	                                                                                      WHERE CHIAVE=dettaglio_spedizioni.TITOLO) 
	                    AND gruppi_case_ed.CHIAVE IN (286071,286072)
			 LIMIT 0,1)              
 ORDER 
    BY GRUPPO_CASA,EDITORE_TITOLO,NOME_TITOLO
